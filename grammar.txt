expression -> literal 
            | unary 
            | binary 
            | grouping;

literal     -> NUMBER | STRING | "true" | "false" | "nil" ;
grouping    -> "(" expresssion ")"
unary       -> ("-" | "!") expression ;
binary      -> expression operator expression ;
operator    -> "==" | "!=" | "<" | "<=" | ">" | ">=" | "+"  | "-"  | "*" | "/" ;
